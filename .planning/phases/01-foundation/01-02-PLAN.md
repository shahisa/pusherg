import { execa } from 'execa'

export async function runScript(packageManager, scriptName, { cwd } = {}) {
  const start = Date.now()
  const proc = await execa(packageManager, ['run', scriptName], {
    reject: false,
    cwd,
    all: true,
  })
  return {
    name: scriptName,
    passed: proc.exitCode === 0,
    exitCode: proc.exitCode,
    stdout: proc.stdout ?? '',
    stderr: proc.stderr ?? '',
    duration: Date.now() - start,
  }
}
```

Key requirements:
- `reject: false` — NEVER let execa throw on non-zero exit
- Accept `cwd` in options object, pass to execa
- `all: true` to capture combined output
- Duration measured with Date.now() delta
- Return exactly the result shape from the interfaces block

import chalk from 'chalk'

export function printResult(result) {
  const icon = result.passed ? chalk.green('\u2713') : chalk.red('\u2717')
  process.stderr.write(`  ${icon} ${result.name} (${result.duration}ms)\n`)
}

export function printSummary(results) {
  const failed = results.filter(r => !r.passed)
  if (failed.length === 0) {
    process.stderr.write(chalk.green('\nAll checks passed.\n'))
  } else {
    process.stderr.write(chalk.red(`\n${failed.length} check(s) failed.\n`))
  }
}
```

Key requirements:
- ALL output via `process.stderr.write()` — NEVER console.log or process.stdout
- chalk.green for pass icon, chalk.red for fail icon
- printResult shows: icon, script name, duration in ms
- printSummary counts failures, shows "All checks passed" or "N check(s) failed"
